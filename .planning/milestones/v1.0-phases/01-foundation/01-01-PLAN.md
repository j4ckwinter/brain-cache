   ```typescript
   export type VRAMTier = 'none' | 'standard' | 'large';

     detectedAt:     z.string().datetime(),
     vramTier:       z.enum(['none', 'standard', 'large']),

   });
   ```


3. **src/services/logger.ts** — Create with pino writing to stderr fd 2 (per D-13, D-14, D-15, D-16):
   ```typescript
   import pino from 'pino';

   const VALID_LEVELS = ['debug', 'info', 'warn', 'error', 'silent'] as const;
   type LogLevel = typeof VALID_LEVELS[number];

   function resolveLevel(): LogLevel {
     const env = process.env.BRAIN_CACHE_LOG?.toLowerCase();
     if (VALID_LEVELS.includes(env as LogLevel)) return env as LogLevel;
     return 'warn';
   }

   export const logger = pino(
     { level: resolveLevel() },
     pino.destination(2) // stderr, always — per D-16
   );

   export function childLogger(component: string) {
     return logger.child({ component });
   }
   ```

4. **tests/services/logger.test.ts** — Write tests covering the behavior items listed above. Mock `pino` and `pino.destination` to verify the destination argument is `2`. Test `resolveLevel` logic by setting/clearing `process.env.BRAIN_CACHE_LOG`. For schema tests, import `CapabilityProfileSchema` and test parse/safeParse with valid and invalid objects.

Per D-14: log level controlled ONLY via `BRAIN_CACHE_LOG` env var. No CLI flags, no config file option.
Per D-15: pino outputs structured JSON to stderr always. No pretty-printing mode.
Per D-16: stdout is strictly reserved for MCP stdio transport.
